127  /******** Analyze spectra *******/

129  bool doFitPeak = config->get("Peak Fitting" , false);

131  bool doFitEfficiency = config->get("Efficiency Fitting" , false);
132  bool doMakePlot = config->get("Make Plots" , false);
133 
134  for(int i = 0; i < (int)spec.size(); ++i) {
135  if(doFindPeak) {
136  MESSAGE("Finding peaks in spectrum'"+filename[i]+"'");
137  double sigma = config->get("Sigma" , 2.);
138  double threshold = config->get("Threshold", 0.2);

149  MESSAGE("Fitting peaks in spectrum'"+filename[i]+"'");

152  TString formula = config->get("Fitting Formula", "gaus+expo");

154  if(min_range == 0 && max_range == 0) {

163  TF1* f = spec[i].fitPeak(min_range,max_range,formula);
164  } else {

166  }
167  }
